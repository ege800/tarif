#include #include #include #include #include #include using namespace std; // ============================= // MAHKEME DEDÜKSİYON OYUNU v1 (C++) // Terminal üzerinden oynanır // ============================= vector ROLES = {"Katil", "Savcı", "Hakim", "Avukat", "Şüpheli"}; struct Player { string name; string role; vector knowledge; bool is_alive = true; }; // Random engine random_device rd; mt19937 rng(rd()); // ----------------------------- // DELİL SİSTEMİ // ----------------------------- vector generate_evidence() { vector evidences = { "Kan izi iki farklı yöne gidiyor", "Parmak izi Şüpheli A'ya ait", "İki farklı ayak izi var", "Telefon logları gece 02:00", "Kredi kartı olaydan sonra kullanılmış", "Görgü tanığı birini koşarken gördü", "Cüzdan olay yerinde yok" }; shuffle(evidences.begin(), evidences.end(), rng); return evidences; } // ----------------------------- // OYUN SINIFI // ----------------------------- class Game { public: vector players; vector evidences; int killerIndex = -1; Game(const vector& names) { evidences = generate_evidence(); assign_roles(names); distribute_knowledge(); } void assign_roles(const vector& names) { vector roles = ROLES; shuffle(roles.begin(), roles.end(), rng); for (size_t i = 0; i < names.size(); i++) { Player p; p.name = names[i]; p.role = roles[i % roles.size()]; if (p.role == "Katil") { killerIndex = i; } players.push_back(p); } } void distribute_knowledge() { uniform_int_distribution<> dist(0, evidences.size() - 1); for (auto &p : players) { for (int i = 0; i < 2; i++) { p.knowledge.push_back(evidences[dist(rng)]); } } // Savcı ekstra bilgi for (auto &p : players) { if (p.role == "Savcı") { p.knowledge.push_back("Şüpheli A benim kardeşim..."); } } // Katil sahte bilgi if (killerIndex != -1) { uniform_int_distribution<> dist2(0, evidences.size() - 1); players[killerIndex].knowledge.push_back("(SAHTE) " + evidences[dist2(rng)]); } } void show_player_info(Player &p) { cout << "\n=== " << p.name << " ===\n"; cout << "Rolün: " << p.role << "\n"; cout << "Bildiklerin:\n"; for (auto &k : p.knowledge) { cout << "- " << k << "\n"; } cin.ignore(); cin.get(); } void discussion_phase() { cout << "\n--- TARTIŞMA FAZI ---\n"; for (auto &p : players) { cout << p.name << " konuşmak için ENTER bas..."; cin.ignore(); cin.get(); } } string voting_phase() { cout << "\n--- OYLAMA ---\n"; map votes; for (auto &voter : players) { cout << "\n" << voter.name << " oy veriyor\n"; for (size_t i = 0; i < players.size(); i++) { cout << i << ": " << players[i].name << "\n"; } int choice; cout << "Kimi suçluyorsun?: "; cin >> choice; string selected = players[choice].name; votes[selected]++; } string accused; int maxVotes = 0; for (auto &v : votes) { if (v.second > maxVotes) { maxVotes = v.second; accused = v.first; } } cout << "\nEn çok oy alan: " << accused << "\n"; return accused; } void court_phase(const string &accused_name) { cout << "\n--- MAHKEME ---\n"; Player* judge = nullptr; for (auto &p : players) { if (p.role == "Hakim") judge = &p; } if (!judge) { cout << "Hakim yok!\n"; return; } cout << "Hakim: " << judge->name << "\n"; cout << "Sanık: " << accused_name << "\n"; char decision; cout << "Suçlu mu? (e/h): "; cin >> decision; Player* accused = nullptr; for (auto &p : players) { if (p.name == accused_name) accused = &p; } if (decision == 'e') { if (accused == &players[killerIndex]) { cout << "\nDOĞRU KARAR! Katil yakalandı!\n"; } else { cout << "\nYANLIŞ! Masum biri cezalandırıldı!\n"; cout << "Gerçek katil: " << players[killerIndex].name << "\n"; } } else { cout << "\nSerbest bırakıldı...\n"; cout << "Gerçek katil: " << players[killerIndex].name << "\n"; } } }; // ----------------------------- // MAIN // ----------------------------- int main() { cout << "=== MAHKEME OYUNU (C++) ===\n"; int count; cout << "Kaç oyuncu var?: "; cin >> count; vector names; for (int i = 0; i < count; i++) { string name; cout << "Oyuncu " << i+1 << " adı: "; cin >> name; names.push_back(name); } Game game(names); cout << "\nBir kişi gece sokakta ölü bulundu...\n"; for (auto &p : game.players) { cout << "\nSıradaki: " << p.name << " (kimse bakmasın!) ENTER..."; cin.ignore(); cin.get(); game.show_player_info(p); } game.discussion_phase(); string accused = game.voting_phase(); game.court_phase(accused); return 0; }
